/*


⭐ 6. Find Minimum Element
📌 Problem:

Write a program to find the smallest element in an array.

📥 Sample Input:
Enter size: 5
Enter elements: 10 25 5 40 15

📤 Sample Output:
Minimum: 5


 */






package Day03_Arrays;

public class iii_Min_Element_In_A_Array {
    public static void main(String[] args){
        // Take the array directly
        int[] arr = {10,25,5,40,15};

        //length of the array
        int length = arr.length;

        // To Find the smallest element in an array

        // First assume min number as first element in an array
        int min = arr[0];

        // Now travel from the first index in the array because 0th index is already used
        for(int i = 1; i<length; i++){

            // Check if current number is smaller or not if smaller update the min
            if(arr[i] < min){
                min = arr[i];
            }
        }

        // Now print the max
        System.out.println(min);
    }
}
