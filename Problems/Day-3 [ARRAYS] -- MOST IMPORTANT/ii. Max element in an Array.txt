/*



📊 2. Find Maximum Element
📌 Problem:

Write a program to find the largest element in an array.

📥 Sample Input:
Enter size: 5
Enter elements: 10 25 5 40 15

📤 Sample Output:
Maximum: 40



 */





package Day03_Arrays;

public class ii_Max_Element_In_A_Array {
    public static void main(String[] args){

        // Take the array directly
        int[] arr = {10,25,5,40,15};

        //length of the array
        int length = arr.length;

        // To Find the largest element in an array

        // First assume max number as first element in an array
        int max = arr[0];

        // Now travel from the first index in the array because 0th index is already used
        for(int i = 1; i<length; i++){

            // Check if current number is greater or not if greater update the max
            if(arr[i] > max){
                max = arr[i];
            }
        }

        // Now print the max
        System.out.println(max);
    }
}
