/*
🔁 5. Palindrome Check
📌 Problem:

Check whether the given string is a palindrome.

Sample I/P = Enter string: madam
Sample O/P = Palindrome
 */


package Basics;

public class v_Palindrome_Check {
    public static void main(String[] args){

        // Initialize the string
        String str = "madam";

        //len of string
        int len = str.length();

        // reverse the string and check if both are equal or not
        StringBuilder reverse = new StringBuilder();

        for(int i = len-1; i>=0; i--){
            // Add each charecter to reverse
            reverse.append(str.charAt(i));
        }


        // Now check if both are equal or not
        if(str.equals(reverse.toString())){
            System.out.println("Palindrome");
        }
        else{
            System.out.println("Not a Palindrome");
        }
    }
}


/*
NOTE -------------> ".equals()" method only for 2 strings  so I converted String Builder to string using "toString()" method
 */
