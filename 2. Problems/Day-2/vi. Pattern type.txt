package Day02;

import java.util.Scanner;

public class vi_Right_Angles_Triangle_Pattern {
    public static void main(String[] args){

        // Pattern problem will get solved by using nested loops

        // Take input from user for no.of rows
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter the number of rows: ");
        int n = sc.nextInt(); // ---> Rows


        for(int i = 0; i <= n; i++){
            for(int j = 0; j < i; j++){
                System.out.print("*");
            }
            System.out.println();
        }
    }
}
