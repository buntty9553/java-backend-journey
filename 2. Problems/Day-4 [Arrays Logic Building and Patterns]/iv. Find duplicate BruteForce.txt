/*



🔍 4. Find Duplicate (Brute Force First)
📌 Problem:

Find any duplicate element in the array

📥 Input:
Enter size: 6

Enter elements: 1 3 4 2 5 3

📤 Output:
Duplicate: 3

👉 Pattern: (think carefully — not obvious)



 */





package Day04_Array_Problem_solving;

public class iv_Find_Duplicate_BruteForce {
    public static void main(String[] args){

        // Take the array directly
        int[] arr = {1,3,4,2,5,3,2};


        /*
        First test the edge cases
        ---> 1. If the array is null
        ---> 2. If the array has no elements
        ---> 3. If the array has only 1 value
         */

        if(arr == null || arr.length <= 1){
            System.out.println("Array is not applicable");
            return;
        }


        /*
        Now come to original part

        Logic :
                1. Use nested for loops because each and every subset want to be compared

         */


        // Initially assume that there is no duplicate in the array
        boolean duplicate = false;

        // Run the loop
        for(int i = 0; i<arr.length; i++){
            for(int j = i+1; j<arr.length; j++){

                // Check the condition
                if(arr[i] == arr[j]){
                    System.out.println("Duplicate: "+ arr[i]);
                    duplicate = true;
                    break;
                }
            }

            // This is to find only first duplicate if there are multiple duplicate values this will not work.
            if(duplicate){
                break;
            }

        }

        if(!duplicate){
            System.out.println("There are no duplicates in the Array.");
        }

    }
}
